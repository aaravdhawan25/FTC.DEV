// Intaker.java — State-machine intake with color distance sensor
@Config
public class Intaker implements Subsystem {

    public DcMotor intake, transfer;
    Telemetry telemetry;
    public double intakePower;
    RevColorSensorV3 disColor;

    public static double intakeOff        = 0;
    public static double intakeIntake     = -1;
    public static double intakeTransfer   = 0.5;
    public static double intakeReverse    = 1;
    public static double transferPowerON  = -1;
    public static double transferPowerOFF = 0;
    public static double transferPowerRev = 1;

    private IntakeStates state = IntakeStates.IDLE;

    public enum IntakeStates {
        INTAKING, INTAKE_ONLY, CHECK, MANUALIN,
        TRANSFER, REVERSE, ReverseWith,
        TRANSFER_SHOOTER, IDLE
    }

    public Intaker(HardwareMap hardwareMap, Telemetry telemetry) {
        this.telemetry = telemetry;
        intake    = hardwareMap.get(DcMotor.class, "intake");
        transfer  = hardwareMap.get(DcMotor.class, "transfer");
        disColor  = hardwareMap.get(RevColorSensorV3.class, "distSensor");
    }

    public double getDist() {
        return disColor.getDistance(DistanceUnit.CM);
    }

    @Override
    public void update() {
        switch (state) {
            case IDLE:
                transfer.setPower(transferPowerOFF);
                intake.setPower(intakeOff);
                break;

            case INTAKING:
                if (getDist() < 3) { setState(IntakeStates.INTAKE_ONLY); }
                intake.setPower(intakeIntake);
                transfer.setPower(transferPowerON);
                break;

            case INTAKE_ONLY:
                if (getDist() > 5) { setState(IntakeStates.INTAKING); }
                intake.setPower(intakeIntake);
                transfer.setPower(transferPowerOFF); // hold — sample detected
                break;

            case MANUALIN:
            case TRANSFER:
            case TRANSFER_SHOOTER:
                intake.setPower(intakeIntake);
                transfer.setPower(transferPowerON);
                break;

            case REVERSE:
                intake.setPower(intakeReverse);
                break;

            case ReverseWith:
                intake.setPower(intakeReverse);
                transfer.setPower(transferPowerRev);
                break;
        }
        telemetry.addData("Intake State", state);
        telemetry.addData("Distance (cm)", getDist());
    }

    @Override
    public void updateCtrls(Gamepad gp1, Gamepad gp2) {
        if (gp1.aWasPressed())            Transfer();
        if (gp1.aWasReleased())           IntakeIdle();
        if (gp1.rightBumperWasPressed())  Intake();
        if (gp1.rightBumperWasReleased()) IntakeIdle();
        if (gp1.leftBumperWasPressed())   Reverse();
        if (gp1.leftBumperWasReleased())  IntakeIdle();
        if (gp1.bWasPressed())            Rev();
        if (gp1.bWasReleased())           IntakeIdle();
    }
}

// Shooter.java — Bang-bang flywheel with hood servo + blocker
@Config
public class Shooter implements Subsystem {

    public static double IDLE_SHOOTER     = 2500;
    public static double CLOSE_SHOOTER_RPM = 3000;
    public static double FAR_SHOOTER_RPM   = 4000;
    public static double AUTO_SHOOTER_RPM  = 2750;
    public static double RPM_TOLERANCE     = 100;
    public static double TICKS_PER_REV     = 28.0;

    public static double targetRPM = 0;

    private ShooterState state = ShooterState.IDLE;

    public enum ShooterState {
        IDLE, SPINNING_UP_CLOSE, SPINNING_UP_FAR, SPINNING_UP_AUTO,
        READY_CLOSE, READY_FAR, READY_AUTO,
        SPINNING_UP_DISTANCE, READY_DISTANCE,
        OPEN_BLOCKER, STOPPING, OFF
    }

    public double getShooterRPM() {
        return (shooterMotor.getVelocity() * 60.0) / TICKS_PER_REV;
    }

    public boolean atTargetSpeed() {
        return Math.abs(targetRPM - getShooterRPM()) < RPM_TOLERANCE;
    }

    // Bang-bang controller — full power until at speed, cut when over
    public void setBangBangPower(double targetRPM) {
        if (getShooterRPM() <= targetRPM) {
            shooterMotor.setPower(1);
            shooterMotor2.setPower(1);
        } else {
            shooterMotor.setPower(0);
            shooterMotor2.setPower(0);
        }
    }

    // Distance-based RPM curve (quadratic regression)
    public double getRPM(double dist) {
        return -0.105746 * dist * dist + 32.33112 * dist + 1558.21332;
    }

    @Override
    public void update() {
        switch (state) {
            case IDLE:
                targetRPM = IDLE_SHOOTER;
                hoodServoPosition = HOOD_SERVO_MIN;
                blockerClose();
                break;
            case SPINNING_UP_CLOSE:
                targetRPM = CLOSE_SHOOTER_RPM;
                hoodServoPosition = HOOD_SERVO_MAX;
                blockerClose();
                if (atTargetSpeed()) setState(ShooterState.READY_CLOSE);
                break;
            case READY_CLOSE:
                targetRPM = CLOSE_SHOOTER_RPM;
                hoodServoPosition = HOOD_SERVO_MAX;
                blockerOpen();
                break;
            case SPINNING_UP_DISTANCE:
                targetRPM = getRPM(CameraConstants.distanceToGoalLL);
                hoodServoPosition = getHoodPos(CameraConstants.distanceToGoalLL);
                if (atTargetSpeed()) setState(ShooterState.READY_DISTANCE);
                break;
            case READY_DISTANCE:
                targetRPM = getRPM(CameraConstants.distanceToGoalLL);
                hoodServoPosition = getHoodPos(CameraConstants.distanceToGoalLL);
                blockerOpen();
                break;
            case STOPPING:
                targetRPM = IDLE_SHOOTER;
                blockerClose();
                if (getShooterRPM() < 100) setState(ShooterState.IDLE);
                break;
        }

        setBangBangPower(targetRPM);
        hoodServo.setPosition(hoodServoPosition);

        telemetry.addData("Shooter State", state);
        telemetry.addData("Current RPM", getShooterRPM());
        telemetry.addData("Target RPM", targetRPM);
        telemetry.addData("At Speed", atTargetSpeed());
    }
}

// Drivetrain.java — Mecanum with slow-mode toggle + LLCam auto-align
@Config
public class Drivetrain implements Subsystem {

    public static DcMotorEx lfMotor, lbMotor, rfMotor, rbMotor;

    double y = 0, x = 0, rx = 0, denominator = 0;
    public static double lfPower, lbPower, rfPower, rbPower;
    public boolean bumpToggle = false;

    public Drivetrain(HardwareMap map, Telemetry telemetry) {
        lfMotor = map.get(DcMotorEx.class, "LFM");
        lbMotor = map.get(DcMotorEx.class, "LBM");
        rfMotor = map.get(DcMotorEx.class, "RFM");
        rbMotor = map.get(DcMotorEx.class, "RBM");
    }

    @Override
    public void init() {
        lbMotor.setDirection(DcMotorEx.Direction.FORWARD);
        rfMotor.setDirection(DcMotorEx.Direction.REVERSE);
        rbMotor.setDirection(DcMotorEx.Direction.REVERSE);
        lfMotor.setDirection(DcMotor.Direction.FORWARD);

        for (DcMotorEx m : new DcMotorEx[]{lfMotor, lbMotor, rfMotor, rbMotor})
            m.setZeroPowerBehavior(DcMotorEx.ZeroPowerBehavior.BRAKE);
    }

    public void setDriveVectors(double y, double x, double rx) {
        this.y = y; this.x = x; this.rx = rx;
    }

    @Override
    public void update() {
        denominator = Math.max(Math.abs(y) + Math.abs(x) + Math.abs(rx), 1);
        lfPower = (y + x + rx) / denominator;
        lbPower = (y - x + rx) / denominator;
        rfPower = (y - x - rx) / denominator;
        rbPower = (y + x - rx) / denominator;

        // Auto-align override via LLCam vision
        if (LLCam.getState() == LLCam.alignState.MATH_CAMERA) {
            lfMotor.setPower(-LLCam.rotationalPower);
            lbMotor.setPower(-LLCam.rotationalPower);
            rfMotor.setPower( LLCam.rotationalPower);
            rbMotor.setPower( LLCam.rotationalPower);
        } else {
            lfMotor.setPower(lfPower);
            lbMotor.setPower(lbPower);
            rfMotor.setPower(rfPower);
            rbMotor.setPower(rbPower);
        }
    }

    @Override
    public void updateCtrls(Gamepad gp1, Gamepad gp2) {
        if (gp1.touchpadWasPressed()) bumpToggle = !bumpToggle;

        if (bumpToggle) {
            // Slow mode — 30 / 40 / 30 % of full speed
            setDriveVectors(-gp1.left_stick_y * 0.3,
                             gp1.left_stick_x  * 0.4,
                             gp1.right_stick_x * 0.3);
        } else {
            setDriveVectors(-gp1.left_stick_y,
                             gp1.left_stick_x,
                             gp1.right_stick_x);
        }
        update();
    }
}

// PedroCloseAuto.java — Full Pedro Pathing autonomous with shooter + intake
public class PedroCloseAuto extends OpMode {

    private Follower   follower;
    private Shooter    shooter;
    private Intaker    intaker;
    private CloseAutoPaths paths;
    private ElapsedTime timer, stateTimer;

    public enum AutoState {
        REVERSE_OUT,  INTAKE_SPIKE_2, RETURN_SPIKE_2,
        GO_TO_GATE,   GATE_INTAKE,    WAIT_GATE_INTAKE, RETURN_GATE,
        INTAKE_SPIKE_1, WAIT_INTAKE_1, RETURN_SPIKE_1,
        INTAKE_SPIKE_3, WAIT_INTAKE_3, RETURN_SPIKE_3,
        SHOOTING_SPINUP, SHOOTING_TRANSFER, DONE
    }

    private AutoState currentState, nextStateAfterShoot;

    @Override
    public void loop() {
        follower.update();
        shooter.update();
        intaker.update();

        switch (currentState) {
            case REVERSE_OUT:
                if (!follower.isBusy()) prepareShoot(AutoState.INTAKE_SPIKE_2);
                break;

            case INTAKE_SPIKE_2:
                if (!follower.isBusy()) setPathState(AutoState.RETURN_SPIKE_2);
                break;

            case RETURN_SPIKE_2:
                if (!follower.isBusy()) {
                    intaker.IntakeIdle();
                    prepareShoot(AutoState.GO_TO_GATE);
                }
                break;

            case WAIT_GATE_INTAKE:
                if (stateTimer.milliseconds() > 700)
                    setPathState(AutoState.RETURN_GATE);
                break;

            case RETURN_GATE:
                if (!follower.isBusy()) prepareShoot(AutoState.INTAKE_SPIKE_1);
                break;

            case SHOOTING_SPINUP:
                // Wait for flywheel to reach target RPM
                if (shooter.getState() == Shooter.ShooterState.READY_DISTANCE) {
                    stateTimer.reset();
                    currentState = AutoState.SHOOTING_TRANSFER;
                }
                break;

            case SHOOTING_TRANSFER:
                if (stateTimer.milliseconds() > 100 &&
                    stateTimer.milliseconds() < 1000) {
                    intaker.TransferShootSub(); // feed ball into shooter
                } else {
                    intaker.IntakeIdle();
                    shooter.stop();
                    setPathState(nextStateAfterShoot);
                }
                break;

            case DONE:
                shooter.stop();
                intaker.IntakeIdle();
                break;
        }
    }

    private void setPathState(AutoState newState) {
        currentState = newState;
        stateTimer.reset();
        switch (newState) {
            case REVERSE_OUT:    follower.followPath(paths.ReverseOut);    break;
            case INTAKE_SPIKE_2: intaker.Intake();
                                 follower.followPath(paths.IntakeSpike2); break;
            case RETURN_SPIKE_2: follower.followPath(paths.ReturnSpike2); break;
            case GO_TO_GATE:     follower.followPath(paths.Gate);          break;
            case GATE_INTAKE:    intaker.Intake();
                                 follower.followPath(paths.GateIntake);   break;
            case RETURN_GATE:    follower.followPath(paths.ReturnGate);    break;
            case INTAKE_SPIKE_1: intaker.Intake();
                                 follower.followPath(paths.IntakeSpike1); break;
            case RETURN_SPIKE_1: follower.followPath(paths.ReturnSpike1); break;
            case INTAKE_SPIKE_3: intaker.Intake();
                                 follower.followPath(paths.IntakeSpike3); break;
            case RETURN_SPIKE_3: follower.followPath(paths.ReturnSpike3); break;
        }
    }

    // Spin up shooter, store next destination, then transition to SHOOTING_SPINUP
    private void prepareShoot(AutoState nextState) {
        nextStateAfterShoot = nextState;
        intaker.IntakeIdle();
        shooter.spinUpDistance();
        currentState = AutoState.SHOOTING_SPINUP;
        stateTimer.reset();
    }

    // Path definitions — BezierLine + BezierCurve with mirrored poses per alliance color
    public static class CloseAutoPaths {
        public PathChain ReverseOut, IntakeSpike2, ReturnSpike2;
        public PathChain Gate, GateIntake, ReturnGate;
        public PathChain IntakeSpike1, ReturnSpike1;
        public PathChain IntakeSpike3, ReturnSpike3;

        public CloseAutoPaths(Follower follower, String color) {
            Pose startPose    = CloseAutoPoseData.mirror(CloseAutoPoseData.START_POSE, color);
            Pose shootingPose = CloseAutoPoseData.mirror(CloseAutoPoseData.SHOOTING_POSE, color);
            Pose secondIntake = CloseAutoPoseData.mirror(CloseAutoPoseData.SECOND_INTAKE, color);
            Pose leverPose    = CloseAutoPoseData.mirror(CloseAutoPoseData.LEVER, color);

            follower.setStartingPose(startPose);

            ReverseOut = follower.pathBuilder()
                .addPath(new BezierLine(startPose, shootingPose))
                .setLinearHeadingInterpolation(/* start → shoot heading */)
                .build();

            IntakeSpike2 = follower.pathBuilder()
                .addPath(new BezierCurve(shootingPose, mid2Curve, secondIntake))
                .setTangentHeadingInterpolation()
                .build();

            Gate = follower.pathBuilder()
                .addPath(new BezierLine(shootingPose, leverPose))
                .setLinearHeadingInterpolation(/* shoot → lever heading */)
                .build();
            // ... remaining paths follow same pattern
        }
    }
}

HardwareMap is the bridge between Java and the physical robot. Every motor, servo, and sensor must be declared in the Robot Controller app's config file, then retrieved by its exact name. Always configure mode, direction, and zero-power behavior in init() before the OpMode starts. 

// Names must match the Robot Controller config file exactly
DcMotorEx      liftMotor   = hw.get(DcMotorEx.class,      "liftMotor");
Servo          clawServo   = hw.get(Servo.class,         "claw");
DistanceSensor distSensor  = hw.get(DistanceSensor.class, "intakeSensor");
TouchSensor    limitSwitch = hw.get(TouchSensor.class,    "limitSwitch");

// Motor configuration
liftMotor.setMode(DcMotor.RunMode.STOP_AND_RESET_ENCODER);
liftMotor.setMode(DcMotor.RunMode.RUN_WITHOUT_ENCODER);
liftMotor.setZeroPowerBehavior(DcMotor.ZeroPowerBehavior.BRAKE);
liftMotor.setDirection(DcMotorSimple.Direction.FORWARD);

// Servo position range: 0.0 (min) → 1.0 (max)
clawServo.setPosition(0.5);

// Read sensor values
double  distCm    = distSensor.getDistance(DistanceUnit.CM);
boolean isPressed = limitSwitch.isPressed();

IMU imu = hw.get(IMU.class, "imu");

// Set orientation to match how the Control Hub is physically mounted
imu.initialize(new IMU.Parameters(
    new RevHubOrientationOnRobot(
        RevHubOrientationOnRobot.LogoFacingDirection.UP,
        RevHubOrientationOnRobot.UsbFacingDirection.FORWARD
    )
));

// Read robot heading every loop
double headingDeg = imu.getRobotYawPitchRollAngles()
                       .getYaw(AngleUnit.DEGREES);

// Rotate stick inputs by heading for field-centric drive
double rad  = Math.toRadians(headingDeg);
double rotX = x * Math.cos(-rad) - y * Math.sin(-rad);
double rotY = x * Math.sin(-rad) + y * Math.cos(-rad);

TeleOp is the driver-controlled period. Gamepad axes return values from −1.0 to 1.0 and triggers from 0.0 to 1.0. Use WasPressed / WasReleased for one-shot edge detection, deadbands to eliminate stick drift, and input curves to give drivers finer control near center. 

@Override
public void updateCtrls(Gamepad gp1, Gamepad gp2) {
    // Analog sticks — up = negative on Y, so negate
    double drive  = -gp1.left_stick_y;
    double strafe =  gp1.left_stick_x;
    double rotate =  gp1.right_stick_x;

    // Triggers: 0.0 (idle) → 1.0 (fully pressed)
    double liftUp   = gp2.right_trigger;
    double liftDown = gp2.left_trigger;

    // WasPressed / WasReleased — fires once per edge, not while held
    if (gp1.rightBumperWasPressed())  intake.Intake();
    if (gp1.rightBumperWasReleased()) intake.IntakeIdle();

    // D-pad for fine-tuning positions
    if (gp2.dpadUp)   lift.nudge(+10);
    if (gp2.dpadDown) lift.nudge(-10);
}

// Simple boolean toggle — flips on each press
private boolean clawOpen = false;

if (gp1.aWasPressed()) {
    clawOpen = !clawOpen;
    clawServo.setPosition(clawOpen ? 1.0 : 0.0);
}

// Multi-position cycle: ground → low → high → ground → ...
private int armPos = 0;
private static final int[] ARM_TICKS = { 0, 480, 1350 };

if (gp2.bWasPressed()) {
    armPos = (armPos + 1) % ARM_TICKS.length; // wraps 0→1→2→0
    lift.setTarget(ARM_TICKS[armPos]);
}

// Speed mode toggle — touchpad switches full / slow speed
private boolean slowMode = false;
if (gp1.touchpadWasPressed()) slowMode = !slowMode;
double speed = slowMode ? 0.3 : 1.0;
setDriveVectors(-gp1.left_stick_y * speed,
                  gp1.left_stick_x * speed,
                  gp1.right_stick_x * speed);

// Deadband — ignore small values to prevent drift
public double deadband(double value, double threshold) {
    if (Math.abs(value) < threshold) return 0;
    // Rescale so output starts at 0 just outside the deadband
    return (value - Math.signum(value) * threshold) / (1.0 - threshold);
}

// Quadratic curve — more precision near center, full power at edges
public double inputCurve(double raw) {
    return Math.signum(raw) * raw * raw;
}

// Apply both to every stick axis before use
double y  = inputCurve(deadband(-gp1.left_stick_y,  0.05));
double x  = inputCurve(deadband( gp1.left_stick_x,  0.05));
double rx = deadband(gp1.right_stick_x, 0.04);

Pedro Pathing uses Bezier curves and heading interpolation for smooth, predictable autonomous movement. A Follower continuously corrects robot position using localization. Define paths from control points, then sequence them with a state machine — calling follower.update() every loop. 

// Initialize follower with your tuned constant classes
Follower follower = new Follower(hardwareMap, FConstants.class, LConstants.class);

Pose start  = new Pose(0,  0,  Math.toRadians(0));
Pose end    = new Pose(48, 24, Math.toRadians(90));
Pose ctrl   = new Pose(24, 0,  0); // Bezier control point (bends the path)

follower.setStartingPose(start);

// Straight line — linear heading interpolation
PathChain line = follower.pathBuilder()
    .addPath(new BezierLine(start, end))
    .setLinearHeadingInterpolation(start.getHeading(), end.getHeading())
    .build();

// Curved path — robot faces direction of travel (tangent)
PathChain curve = follower.pathBuilder()
    .addPath(new BezierCurve(start, ctrl, end))
    .setTangentHeadingInterpolation()
    .build();

follower.followPath(line);

// State machine autonomous — setPathState() starts a path + resets timer
private void setPathState(AutoState newState) {
    currentState = newState;
    stateTimer.reset();
    switch (newState) {
        case DRIVE_TO_SPIKE:
            intaker.Intake();
            follower.followPath(paths.ToSpike);
            break;
        case RETURN_TO_SHOOT:
            intaker.IntakeIdle();
            follower.followPath(paths.ToShoot);
            break;
    }
}

@Override
public void loop() {
    follower.update(); // always call every loop

    switch (currentState) {
        case DRIVE_TO_SPIKE:
            if (!follower.isBusy()) setPathState(AutoState.WAIT_FOR_SAMPLE);
            break;
        case WAIT_FOR_SAMPLE:
            if (stateTimer.milliseconds() > 500)
                setPathState(AutoState.RETURN_TO_SHOOT);
            break;
    }

    // FTC Dashboard — live path + robot visualization
    Drawing.drawRobot(follower.poseUpdater.getPose(), "#4CAF50");
    Drawing.sendPacket();
}

The FTC SDK's VisionPortal API provides built-in AprilTag detection — no external libraries needed. AprilTags are fiducial markers that return precise 3D pose data: range (distance), bearing (horizontal angle), and yaw (rotation). Works with any USB webcam supported by the SDK. 

// Build the AprilTag processor
AprilTagProcessor aprilTag = new AprilTagProcessor.Builder()
    .setDrawAxes(true)
    .setDrawCubeProjection(true)
    .setDrawTagOutline(true)
    .setDrawTagID(true)
    .build();

// Attach processor to a VisionPortal
VisionPortal visionPortal = new VisionPortal.Builder()
    .setCamera(hardwareMap.get(WebcamName.class, "Webcam 1"))
    .addProcessor(aprilTag)
    .setCameraResolution(new Size(640, 480))
    .build();

// Pause/resume streaming to save CPU when vision not needed
visionPortal.stopStreaming();
visionPortal.resumeStreaming();

// Poll detections every loop iteration
List<AprilTagDetection> detections = aprilTag.getDetections();

for (AprilTagDetection tag : detections) {
    if (tag.metadata != null) {
        // Pose relative to camera — in inches and degrees
        double range   = tag.ftcPose.range;   // distance to tag
        double bearing = tag.ftcPose.bearing; // left/right angle
        double yaw     = tag.ftcPose.yaw;     // tag rotation

        telemetry.addData("Tag ID",      tag.id);
        telemetry.addData("Range (in)",  String.format("%.1f", range));
        telemetry.addData("Bearing (°)", String.format("%.1f", bearing));

        // Simple P-controller to align: rotate until bearing ≈ 0
        double turnPower = bearing * 0.02;
        drivetrain.setDriveVectors(0, 0, turnPower);
    }
}

PID (Proportional–Integral–Derivative) produces a correction based on current error, accumulated error, and rate of change. PIDF adds a feedforward term — a constant force in the direction of the target, essential for lifts fighting gravity. FTCLib provides ready-to-use controllers; custom versions give full tuning control. 

// FTCLib PIDController — import com.arcrobotics.ftclib.controller.PIDController
PIDController pid = new PIDController(0.008, 0.0, 0.0004);
//                                       kP      kI    kD

pid.setSetPoint(targetTicks); // desired encoder position
pid.setTolerance(15);        // acceptable error in ticks

// Call every loop iteration
public void update() {
    double current = motor.getCurrentPosition();
    double output  = pid.calculate(current);
    motor.setPower(MathUtils.clamp(output, -1.0, 1.0));
}

boolean done = pid.atSetPoint(); // true when within tolerance

// FTCLib PIDFController — adds feedforward (kF) term
// import com.arcrobotics.ftclib.controller.PIDFController
PIDFController pidf = new PIDFController(0.008, 0.0, 0.0004, 0.0002);
//                                          kP      kI    kD      kF

// output = kP*error + kI*∑error + kD*Δerror + kF*setpoint
// kF counteracts gravity — tune it to the power needed to hold position still

public void update() {
    double current = motor.getCurrentPosition();
    // calculate(measured, setpoint) — two-arg form sets target inline
    double output  = pidf.calculate(current, targetTicks);
    motor.setPower(MathUtils.clamp(output, -1.0, 1.0));
}

// Custom PID — no FTCLib dependency required
public class PIDController {
    private double kP, kI, kD;
    private double integral  = 0;
    private double lastError = 0;
    private long   lastTime  = System.nanoTime();

    public PIDController(double kP, double kI, double kD) {
        this.kP = kP; this.kI = kI; this.kD = kD;
    }

    public double calculate(double current, double target) {
        long   now        = System.nanoTime();
        double dt         = (now - lastTime) / 1e9;
        lastTime           = now;
        double error      = target - current;
        integral          += error * dt;
        double derivative = dt > 0 ? (error - lastError) / dt : 0;
        lastError          = error;
        return kP * error + kI * integral + kD * derivative;
    }

    public void reset() { integral = 0; lastError = 0; }
}

// Custom PIDF — extends the PID above with a feedforward term
public class PIDFController extends PIDController {
    private double kF;

    public PIDFController(double kP, double kI, double kD, double kF) {
        super(kP, kI, kD);
        this.kF = kF;
    }

    @Override
    public double calculate(double current, double target) {
        double pid         = super.calculate(current, target);
        double feedforward = kF * target; // force proportional to target position
        return pid + feedforward;
    }
}

// Usage — vertical lift fighting gravity
PIDFController lift = new PIDFController(0.008, 0.0, 0.0004, 0.00015);
double power = lift.calculate(motor.getCurrentPosition(), targetTicks);
motor.setPower(Math.max(-1, Math.min(1, power)));

FTC has two base classes. LinearOpMode runs sequentially in one runOpMode() method — ideal for simple autonomous where you want to block and wait. OpMode (iterative) splits into init(), loop(), and lifecycle hooks — better for TeleOp and complex autonomous with subsystems and state machines. Most competitive teams use iterative OpMode for both modes. 

@TeleOp(name = "My TeleOp", group = "Main")
public class MyTeleOp extends LinearOpMode {

    @Override
    public void runOpMode() {
        // init — runs once, before PLAY is pressed
        DcMotorEx motor = hardwareMap.get(DcMotorEx.class, "motor");

        telemetry.addData("Status", "Ready");
        telemetry.update();

        waitForStart(); // blocks here until driver presses ▶ PLAY

        // loop — runs until STOP is pressed
        while (opModeIsActive()) {
            motor.setPower(-gamepad1.left_stick_y);
            telemetry.addData("Power", motor.getPower());
            telemetry.update();
        }
    }
}

@TeleOp(name = "My Iterative", group = "Main")
public class MyIterative extends OpMode {
    private DcMotorEx motor;

    @Override
    public void init() {
        // Runs once when Init is pressed
        motor = hardwareMap.get(DcMotorEx.class, "motor");
    }

    @Override
    public void init_loop() {
        // Optional — runs repeatedly between Init and Start
    }

    @Override
    public void start() {
        // Optional — runs once when Start is pressed
    }

    @Override
    public void loop() {
        // Runs repeatedly after Start — all robot logic goes here
        motor.setPower(-gamepad1.left_stick_y);
        telemetry.addData("Power", motor.getPower());
        telemetry.update();
    }

    @Override
    public void stop() {
        // Runs once when Stop is pressed
        motor.setPower(0);
    }
}
